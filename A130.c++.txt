#include <iostream>
#include <stack>
using namespace std;

int main() {
  int k, n, sum=0;
  stack<int> s;
  
  cin >> k;
  for(int i=0; i<k; i++) {
    cin >> n;

    if(n==0) {
      s.pop();
    } else {
      s.push(n);
    }
  }
  
  while(true) {
    if(s.empty()) {
      break;
    } else {
      sum += s.top();
      s.pop();
    }
  }
  cout << sum;
}
